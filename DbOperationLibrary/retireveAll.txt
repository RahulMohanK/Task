Create Procedure retireveAll
    @StaffType int
AS
BEGIN
    IF @StaffType = 0
    BEGIN
        SELECT *
        FROM Staff as S INNER JOIN AdministrativeStaff as Ad ON S.Id = Ad.StaffId;
    END
    IF @StaffType = 1
    BEGIN
        SELECT *
        FROM Staff as S INNER JOIN TeachingStaff as T ON S.Id = T.StaffId;
    END
    IF @StaffType = 2
    BEGIN
        SELECT *
        FROM Staff as S INNER JOIN SupportingStaff as Sp ON S.Id = Sp.StaffId;
    END

END
